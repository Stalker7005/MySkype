project(ClientApp)
cmake_minimum_required(VERSION 3.0)

set(ClientHeaders
	Client.h
	)
set(ClientSources
	Client.cpp
	)

message(STATUS "${PROJECT_NAME} added: ${ClientHeaders} ${ClientSources}")
add_executable(${PROJECT_NAME} ${ClientHeaders} ${ClientSources})

source_group("Client\\Headers" FILES ${ClientHeaders})
source_group("Client\\Sources" FILES ${ClientSources})

include_directories(
					${Boost_INCLUDE_DIRS}
					${LOG4CPLUS_INCLUDE}
					../ODBCHelperLib
					../NetworkLib
                    ../SerializationLib
                    ../BaseObjects
                    ../ThreadsUtilsLib
				    )

message(STATUS "Boost Libraries:${Boost_LIBRARIES}")
target_link_libraries(
					  ${PROJECT_NAME} 
					  ${Boost_LIBRARIES}
					  ${LOG4CPLUS_LIB_DIRS}
					  ODBCHelperLib
					  NetworkLib
                      SerializationLib
                      BaseObjects
                      ThreadsUtilsLib
					  )
